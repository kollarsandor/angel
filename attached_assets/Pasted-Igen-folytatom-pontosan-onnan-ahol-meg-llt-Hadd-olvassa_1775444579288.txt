# 4b. `src/core_relational/verified_inference_engine.zig` — `performVerifiedInferenceWithZK` (folytatás)

A 324-351. sor közötti réteg-ciklus jelenleg `tanh(sum)` aktivációt használ: [1](#7-0) 

**Cseréld le a teljes `while (layer < weights_s.len)` ciklust erre:**

```zig
        var layer: usize = 0;
        while (layer < weights_s.len) : (layer += 1) {
            // RSF affin coupling forward (tanh HELYETT)
            const half = output_buf.len / 2;
            if (half == 0) break;

            // 1. Scale: s = exp(clip(W_s · x2))
            var out_idx: usize = 0;
            while (out_idx < half) : (out_idx += 1) {
                var sum: f32 = 0.0;
                var in_idx: usize = 0;
                while (in_idx < half and in_idx < weights_s[layer].len) : (in_idx += 1) {
                    if (out_idx < weights_s[layer][in_idx].len) {
                        sum += intermediate[in_idx + half] * weights_s[layer][in_idx][out_idx];
                    }
                }
                const clipped = if (sum < -5.0) @as(f32, -5.0) else if (sum > 5.0) @as(f32, 5.0) else sum;
                const scale = @exp(clipped);
                intermediate[out_idx] *= scale;
            }

            // 2. Translation: t = W_t · y1, majd x2 += t
            out_idx = 0;
            while (out_idx < half) : (out_idx += 1) {
                var sum: f32 = 0.0;
                var in_idx: usize = 0;
                while (in_idx < half and in_idx < weights_t[layer].len) : (in_idx += 1) {
                    if (out_idx < weights_t[layer][in_idx].len) {
                        sum += intermediate[in_idx] * weights_t[layer][in_idx][out_idx];
                    }
                }
                intermediate[out_idx + half] += sum;
            }
        }
```

A lényeg: `std.math.tanh(sum)` → `@exp(clipped) * x1` (affin coupling scale), és a második `weights_t` lépés is affin coupling translation lesz, nem egyszerű mátrixszorzás + összeadás.

    (t_weight: [d_model]f32)

    let clip_max = 5.0f32
    in map2 (\s_row t_row ->
      let x1 = take half s_row :> [d_model/2]f32
      let x2 = drop half t_row :> [d_model/2]f32
      let s_w1 = take half s_weight :> [d_model/2]f32

        let raw = xi * wi
        let clipped = f32.max clip_min (f32.min clip_max raw)
        in f32.exp clipped

      let y2 = map2 (+) x2 trans

let softmax [n] (x: [n]f32): [n]f32 =
  if n == 0 then x
  else
    let clean = map (\xi -> if f32.isnan xi then -f32.inf else xi) x
    let max_val = reduce f32.max (-f32.inf) clean
    in if f32.isinf max_val && max_val < 0f32
       then replicate n (1f32 / f32.i64 n)
       else
         let exp_x = map (\xi -> f32.exp (xi - max_val)) clean
         let total = reduce (+) 0f32 exp_x
         in if total <= 0f32 || f32.isnan total
            then replicate n (1f32 / f32.i64 n)
            else map (\e -> e / total) exp_x

let layer_norm [n] (x: [n]f32) (gamma: [n]f32) (beta: [n]f32) (eps: f32): [n]f32 =
  if n == 0 then x
  else
    let safe_eps = f32.max eps 1e-12f32
    let clean = map (\xi -> if f32.isnan xi then 0f32 else xi) x
    let mean_val = (reduce (+) 0f32 clean) / f32.i64 n
    let centered = map (\xi -> xi - mean_val) clean
    let variance = (reduce (+) 0f32 (map (\c -> c * c) centered)) / f32.i64 n
    let std_dev = f32.sqrt (f32.max variance 0f32 + safe_eps)
    in map3 (\c g b -> g * (c / std_dev) + b) centered gamma beta

let relu [n] (x: [n]f32): [n]f32 =
  map (\xi -> if f32.isnan xi then 0f32 else f32.max 0f32 xi) x

let gelu [n] (x: [n]f32): [n]f32 =
  let sqrt_2_over_pi = 0.7978845608028654f32
  in map (\xi ->

      map2 (\o xi -> o + xi * residual_scale) out_row inp_row
